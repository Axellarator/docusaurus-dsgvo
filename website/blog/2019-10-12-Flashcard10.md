function updatescore(){// Richtig und Falsch kalkuliert und als Prozentangabe

		scoreref.innerHTML      = "Ergebnis: " + currentscore   + " / " + total;

	}
}

function pruefung(box, nr, cbox, antw){

			}

		case 2 :
			if(box) {

			} else {

				document.getElementById("field2").checked = false;
